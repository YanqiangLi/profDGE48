#!/local/bin/python

'''
----------------
spikedata_cluster_comparison.py

Takes the spike-in data output from getSpikeCounts.py and the gene data output
from getGeneCounts.py, normalizes the datasets by the total read count in each
lane, clusters the datasets independently and then compares the quality of the
clustering for the two datasets. The script will loop through the samples
removing each one and assessing the impact this has the quality of the
clustering in each cases. Replicates whose removal significantly improves the
clustering of the spike-ins, relative to a threshold value and relative to the
associated change in clustering quality in the gene data, are flagged as 'bad 
spike-in replicates' to be removed when doing downstream analysis that relies 
on normalizations based on the spike-in data.

.. moduleauthor:: Nick Schurch <nschurch@dundee.ac.uk>

Command-line Arguments
======================

**usage\:** 
    spikedata_cluster_comparison.py
    :option:`-g|--genedata` *<file>*      
    :option:`-s|--spikedata` *<file>*
    :option:`-o|--outfile` *<file>*      
    :option:`-l|--logfile` *<file>* 
    [:option:`-m|--metric` *<str>*]
    [:option:`-t|--sigthresh` *<float>*]
    [:option:`-d|--plotdir` *<path>*]
    [:options:`--displayplots`]
    [:option:`-f|--suffix` *<str>*] 
    [:option:`-v|--verbose`] 
    [:option:`--version`] 
    [:option:`--help`]

Required Parameters
-------------------

:option:`--genedata|-g`          

  Path to the .pkl data file output from getGeneCounts.py.

:option:`--spikedata|-s`          

  Path to the .pkl data file output from getSpikeCounts.py.

:option:`--outfile|-o`           

  The name (inc. path) of the output file from the wrapper.

:option:`--logfile|-l`        

  The name (inc. path) of the log file from the wrapper.

Other options
-------------

:option:`--metric|-m`

  The name of the distance metric to use for the clustering. Accepts numpy
  distance metric names. See http://docs.scipy.org/doc/scipy/reference/
  generated/scipy.spatial.distance.pdist.html#scipy.spatial.distance.pdist
  Defaults to 'correlation'.

:option:`--sigthresh|-t`

  The sigma-significance to use when trimming samples. The higher the number,
  the larger the effect that removing a sample needs to for it to be rejected.
  Defaults to 2.

:option:`--plotdir|-d`
  
  Specify a location for the output plots to be saved (if --displayplots is
  False). Defaults to the current working directory.

:option:`--displayplots`

  Turn on interactive plotting instead of saving the figures.

:option:`--suffix|-f`           

  A prefix for the output plots. Defaults to None.

:option:`--help|-h`

  Print a basic description of the tool and its options to STDOUT.

:option:`--version`    

  Show program's version number and exit.
    
:option:`--verbose|-v`     

  Turn on verbose logging (recommended).


Output
======


A pkl (python data structure that uses the cPickle library to read python
objects directly) containing a tuple withh te results form the replicate 
selection process. This tuple also includes the spike in information 
from the file specified in --spikedata.

If --displayplots is used this script plots the graphs to the screen. If not 
specified, it outputs a series of plots suffixed with the string specified by 
--suffix in the location specified by --plotdir.

'''

[docs]def get_colors(num_colors, lightbase=50, satbase=90):
    
    ''' define #num_colors random colours for a given lightness & saturation ''' 
    
    colors=[]
    for i in numpy.arange(0., 360., 360. / num_colors):
        hue = i/360.
        lightness = (lightbase + numpy.random.rand() * 10)/100.
        saturation = (satbase + numpy.random.rand() * 10)/100.
        colors.append(colorsys.hls_to_rgb(hue, lightness, saturation))
    return colors



[docs]def getConditions(axes, axis_label="x"):
    
    ''' returns a dictionary of conditions with a list of the data-cube 
    indexes of samples for that condition'''
    
    conditions = {}
    i=0
    while i<len(axes[axis_label]):
        match = re.match("^(.+)_([0-9]+)", axes[axis_label][i])
        try:
            conditions[match.group(1)].append(i)

[docs]def pearsonCorr(x,y, nsig=1.96):
    
    ''' computes pearsons correlation coefficient with confidence intervals '''
    
    # I really can't be arsed to comment this: read mareks book... its straight 
    # out of there
    
    nsig = float(nsig)
    
    # note, the scipy pearsonr code will produce runtime warnings is either 
    # array has a standard deviation of zero (i.e. is all the same number)!
    
    r = pearsonr(x,y)
    zp = 0.5*numpy.log((1+r[0])/(1-r[0]))
    
    # we don't want any zeros, or negative numbers in len(x)-3; if there are 
    # the sqrt complains and we may get a div by zero error, so in these cases
    # I fix sigp==1
    if len(x)>3:
        sigp = 1.0/numpy.sqrt(len(x)-3)
    else:
        sigp=1.0
    zlow_95 = zp-(nsig*sigp)
    zup_95 = zp+(nsig*sigp)
    rlow = ((numpy.e**(2*zlow_95))-1)/((numpy.e**(2*zlow_95))+1)
    rup = ((numpy.e**(2*zup_95))-1)/((numpy.e**(2*zup_95))+1)

    return(r[0], r[1], rlow, rup)

    rtup = pearsonCorr(gdist, sdist, options.sigthresh)

    samples = numpy.arange(numpy.shape(thisspikedata)[0])
    sampleIDs = numpy.array(sampleIDs)
    
    # report the starting position to the log - note the log is not closed here 
    # cos I'll be writing to it a lot!
    
    corrstr = "%.2f (ul: %.4f, ll:%.4f)" % (rtup[0], rtup[3], rtup[2])
    log = open(options.log,"a")
    log.write("\n\t %s: Initial stats:\n\t\tNumber of samples: %i\n\t\t" \

              )

    # biggest improvement in the corellation between the distance matricies. If 
    # this improvement is bigger that --sigthresh, reject it and contine. If

    removed=[]
    cont = True
    figs=[]
    while cont:
            
        nsamples = len(samples)
        candidates=[]
        
        if options.verbose:
            log.write("\t\tCurrent # of samples & Ids: %i, %i\n" \
                      "" % (len(samples), len(sampleIDs)))
        
        i=0
        while i<nsamples:
            
            # these two lines, remove a specific replicate from the array
            
            split = numpy.split(samples, [i,i+1])
            joined = numpy.array(numpy.concatenate((split[0],
                                                    split[2])
                                                   ),
                                 dtype='int'
                                 )
            
            # compute the new distance matricies without this replicate, and 
            # their correlation
            
            thisgdist = pdist(thisgenedata[joined,], options.distmetric)
            thissdist = pdist(thisspikedata[joined,], options.distmetric)
            thisrtup = pearsonCorr(thisgdist, thissdist, options.sigthresh)
            
            # compute the change in correlation. If the removal is an 
            # improvement (of any size, at this stage) over the starting 
            # position then store it and the information as a candidate for 
            # removal
            
            deltaR = thisrtup[0]-rtup[0]
            if thisrtup[0]>rtup[3]:
                candidates.append([i, deltaR, rtup, thisrtup, sampleIDs[i]])
            
            i+=1

                    maxcandidate = candidate

[docs]def getLaneTotalCountNorm(spike_cube, gene_cube, axes, options):
    
    ''' Calculate the total lane counts, and normalization factors
    
    Including gene data, intergenic data unmapped reads, ambiguous mappings, etc, 
    etc
    '''
    
    log = open(options.log,"a")
    log.write("\n %s: Computing total lane counts and spike-in read " \
              "fraction..." % timeStr())
    log.close()
    
    lane_totcounts={}
    countarray=[]
    i=0
    while i<len(axes["y"]):
        
        # select data just for that lane
        
        spikes = spike_cube[:,i,:]
        genes = gene_cube[:,i,:]
        
        # sum over all samples, and then over all spikeins/genes
        
        spikecounts = numpy.sum(spikes)
        genecounts = numpy.sum(genes)
        totcounts = spikecounts + genecounts
        spike_frac = float(spikecounts)/totcounts
        
        if options.verbose:
            log = open(options.log,"a")
            log.write("\n\t%i: lane %s total counts: %i" \
                      "" % (i, axes["y"][i],
                            totcounts
                            )
                      )
            log.write("\n\t\tdata cube counts: %i (%.4f)" % (spikecounts,
                                                            spike_frac
                                                            )
                      ) 
        
        lane_totcounts[axes["y"][i]] = {"lanecount":totcounts}
        countarray.append(totcounts)
        i+=1
    
    # get the mean and normalization factors to the mean
    
    countarray = numpy.array(countarray)
    norm_factors = countarray/countarray.mean()
    i=0
    while i<len(norm_factors):
        lane_totcounts[axes["y"][i]]["norm_factor"] = norm_factors[i]
        i+=1
    
    return(lane_totcounts)

        i=0
