'''
----------------
bySample_profiles.py

Takes the Spike-in data output from getSpikeCounts.py and makes plots for the 
details of the spike-in variations by biological sample.

.. moduleauthor:: Nick Schurch <nschurch@dundee.ac.uk>

Command-line Arguments
======================

**usage\:** 
    bySample_profiles.py 
    :option:`-g|--genedata` *<file>*      
    :option:`-s|--spikedata` *<file>*
    :option:`-l|--logfile` *<file>* 
    [:option:`-d|--plotdir` *<path>*]
    [:options:`--displayplots`]
    [:option:`-f|--suffix` *<str>*] 
    [:option:`-v|--verbose`] 
    [:option:`--version`] 
    [:option:`--help`]

Required Parameters
-------------------

:option:`--genedata|-g`          

  Path to the .pkl data file output from getGeneCounts.py.

:option:`--spikedata|-s`          

  Path to the .pkl data file output from getSpikeCounts.py.

:option:`--logfile|-l`        

  The name (inc. path) of the log file from the wrapper.

Other options
-------------

:option:`--plotdir|-d`
  
  Specify a location for the output plots to be saved (if --displayplots is
  False). Defaults to the current working directory.

:option:`--displayplots`

  Turn on interactive plotting instead of saving the figures.

:option:`--suffix|-f`           

  A prefix for the output plots. Defaults to None.

:option:`--help|-h`

  Print a basic description of the tool and its options to STDOUT.

:option:`--version`    

  Show program's version number and exit.
    
:option:`--verbose|-v`     

  Turn on verbose logging (recommended).


Output
======

If --displayplots is used this script plots the graphs to the screen and
produces no output. If not specified, it outputs a series of plots suffixed with
the string specified by --suffix in the location specified by --plotdir.

'''

[docs]def get_colors(num_colors, lightbase=50, satbase=90):
    
    ''' define #num_colors random colours for a given lightness & saturation ''' 
    
    colors=[]
    for i in numpy.arange(0., 360., 360. / num_colors):
        hue = i/360.

[docs]def getLaneTotalCountNorm(spike_cube, gene_cube, axes, options):
    
    ''' Calculate the total lane counts, and normalization factors
    
    Including gene data, intergenic data unmapped reads, ambiguous mappings, etc, 
    etc
    '''
    
    log = open(options.log,"a")
    log.write("\n %s: Computing total lane counts and spike-in read " \
              "fraction..." % timeStr())
    log.close()
    
    lane_totcounts={}
    countarray=[]
    i=0
    while i<len(axes["y"]):
        
        # select data just for that lane
        
        spikes = spike_cube[:,i,:]
        genes = gene_cube[:,i,:]
        
        # sum over all samples, and then over all spikeins/genes
        
        spikecounts = numpy.sum(spikes)
        genecounts = numpy.sum(genes)
        totcounts = spikecounts + genecounts
        spike_frac = float(spikecounts)/totcounts
        
        if options.verbose:
            log = open(options.log,"a")
            log.write("\n\t%i: lane %s total counts: %i" \
                      "" % (i, axes["y"][i],
                            totcounts
                            )
                      )
            log.write("\n\t\tspike-in counts: %i (%.4f)" % (spikecounts,
                                                            spike_frac
                                                            )
                      ) 
        
        lane_totcounts[axes["y"][i]] = {"lanecount":totcounts}
        countarray.append(totcounts)
        i+=1
    
    # get the mean and normalization factors to the mean
    
    countarray = numpy.array(countarray)
    norm_factors = countarray/countarray.mean()
    i=0
    while i<len(norm_factors):
        lane_totcounts[axes["y"][i]]["norm_factor"] = norm_factors[i]
        i+=1
    
    return(lane_totcounts)



[docs]def convertRPKM(spike_cube, spike_dic, spike_cube_axes, options):
    
    ''' converts a read count data-cube to a RPKM data cube '''
    
    if options.verbose:    
        log = open(options.log,"a")
        log.write("\n %s: Converting raw counts to RPKM..." % timeStr())
        log.close()
    
    new_cube = numpy.copy(spike_cube)
    i=0
    while i<len(spike_cube_axes["z"]):
        spike = spike_cube_axes["z"][i]
        spikekb = len(spike_dic[spike]["Sequence"])/float(1000)        
        new_cube[:,:,i] = spike_cube[:,:,i]/spikekb
        i+=1

    return(new_cube)



[docs]def normByLanes(spike_cube, axes, options, lanenorms=None, gene_cube=None):
    
    ''' normalize a data-cube by lane normalization factors ''' 

    if lanenorms is None and gene_cube is not None:
        lanenorms = getLaneTotalCountNorm(spike_cube, gene_cube,
                                          axes, options
                                          )
    
    new_cube = numpy.copy(spike_cube) 
    if lanenorms is None:    
        log = open(options.log,"a")
        log.write("****Normalizing by lanes incorrecly called. Doing " \
                  "nothing! ****")
        log.close()
    else:
        if options.verbose:
            log = open(options.log,"a")
            log.write("\n %s: Normalizing by lane total count..." % timeStr())
            log.close()
           
        i=0
        while i<len(axes["y"]):
    
            laneid = axes["y"][i]
            norm_factor = lanenorms[laneid]["norm_factor"]
            lane_count = lanenorms[laneid]["lanecount"]
            
            if options.verbose:
                 log = open(options.log,"a")
                 log.write("\n\t\tLane %s total count: %i\n\t\tNormalization " \
                           "factor: %.2f..." % (laneid, lane_count, norm_factor)
                           )
                 log.close()
            
            new_cube[:,i,:] = new_cube[:,i,:]/norm_factor
            i+=1
    
    return(new_cube)

    data_cube = data[2]
    data_cube_axes = data[3]
    batch_dic = data[4] 

    log = open(options.log,"a")
    log.write("\n %s: Reading gene data file %s..." % (timeStr(),
                                                       options.genedata)
              )
    log.close()    

    file = open(options.genedata, "r")
    gene_data_dic, gene_cube, gene_cube_axes, unmapped_list = cPickle.load(file)
    file.close()
    
    # get total counts per lane and normalization factors
    
    lanenorms = getLaneTotalCountNorm(data_cube, gene_cube,
                                      data_cube_axes, options
                                      )
    
    # plot figures for all samples. This makes a lot of figures, which is fine
    # if they're being written to disk, but less fine when they appear on the 
    # desktop in windows. So, when --displayplots is specified, just make one 
    # example plot.  

    if options.displayplots:        
        log = open(options.log,"a")
        log.write("\n %s: Making mean vs standard deviation plots for 1 " \
                  "example sample. To see these plots for all samples, " \
                  "please run this script without --displayplots." \
                  "" % timeStr())
        log.close()
    else:
        log = open(options.log,"a")
        log.write("\n %s: Making mean vs standard deviation plots for %i " \
                  "samples..." % (timeStr(), len(data_cube_axes["x"])))
        log.close()
    
    allfigs=[]
    i=0
    while i<len(data_cube_axes["x"]):
        fig = plot_muvsig(i, data_cube, data_cube_axes, options)
        fig2 = plot_muvsig(i, data_cube, data_cube_axes, options,
                           RPKM=True, RPKM_dic=spike_dic
                           )
        fig3 = plot_muvsig(i, data_cube, data_cube_axes, options,
                           countnorm=lanenorms
                           )
        fig4 = plot_muvsig(i, data_cube, data_cube_axes, options,
                           countnorm=lanenorms, RPKM=True, RPKM_dic=spike_dic
                           )
        allfigs.append(fig)
        allfigs.append(fig2)
        allfigs.append(fig3)
        allfigs.append(fig4)
        if options.displayplots:
            i=len(data_cube_axes["x"])
        else:
            i+=1
    
    log = open(options.log,"a")
    log.write("\n %s: Applying lane total count normalization..." % timeStr())
    log.close()    
    
    # normalize by total lane counts (this is right so we don't prove a option 
    # for it)
    
    data_cube = normByLanes(data_cube, data_cube_axes, options, 
                            lanenorms=lanenorms)
    
    # make other plots
        
    allfigs = allfigs + plotSampleProfiles(data_cube, data_cube_axes, options)
        
    allfigs = allfigs + plotSampleProfiles(data_cube, data_cube_axes, options,
                                           batch=batch_dic)
    
    allfigs = allfigs + plotSampleClustering(data_cube, data_cube_axes, 
                                             options, batch=batch_dic
                                            )
    
    allfigs = allfigs + plotSampleClustering(data_cube, data_cube_axes, 
                                             options, batch=batch_dic, 
                                             dist="correlation"
                                             )
    
    allfigs = allfigs + plotCorrelationMatrix(data_cube, data_cube_axes,
                                              options)
    
    if options.displayplots:
        plt.show()
    
    log = open(options.log,"a")
    log.write("\n %s: finished." % timeStr())
    log.close()
